Get your Claude setup-token

If you have a Claude Pro or Max subscription, you can generate a setup token to authenticate without an API key:

  1. Open a terminal and install Claude CLI: npm install -g @anthropic-ai/claude-code
  2. Run: claude setup-token

1 Create GitHub Repo (3 Files)

Create a new GitHub repo (e.g., openclaw-railway) with these 3 files:

Dockerfile

# Install OpenClaw from npm for Railway
# Pulls latest stable release automatically

FROM node:22-bookworm

# Install OpenClaw globally
RUN npm install -g openclaw@latest

# Install utilities
RUN apt-get update && apt-get install -y --no-install-recommends jq openssl && \
    rm -rf /var/lib/apt/lists/*

# Create persistent directories
RUN mkdir -p /data/.openclaw /data/workspace

# Copy startup script
COPY railway-entrypoint.sh /app/railway-entrypoint.sh
RUN chmod +x /app/railway-entrypoint.sh

# Defaults
ENV NODE_ENV=production
ENV OPENCLAW_STATE_DIR=/data/.openclaw
ENV OPENCLAW_WORKSPACE_DIR=/data/workspace

EXPOSE ${PORT:-8080}
ENTRYPOINT ["/app/railway-entrypoint.sh"]

railway.toml

[build]
dockerfilePath = "Dockerfile"

[deploy]
healthcheckPath = "/healthz"
healthcheckTimeout = 300
restartPolicyType = "ON_FAILURE"
restartPolicyMaxRetries = 5

railway-entrypoint.sh Copy this exactly - all fixes are baked in

#!/usr/bin/env bash
set -euo pipefail

STATE_DIR="${OPENCLAW_STATE_DIR:-/data/.openclaw}"
WORKSPACE_DIR="${OPENCLAW_WORKSPACE_DIR:-/data/workspace}"
CONFIG_FILE="${STATE_DIR}/openclaw.json"
PORT="${PORT:-8080}"

# Auto-generate gateway token if not set
if [ -z "${OPENCLAW_GATEWAY_TOKEN:-}" ]; then
  if [ -f "${STATE_DIR}/.gateway-token" ]; then
    OPENCLAW_GATEWAY_TOKEN=$(cat "${STATE_DIR}/.gateway-token")
  else
    OPENCLAW_GATEWAY_TOKEN=$(openssl rand -hex 32)
  fi
fi

mkdir -p "${STATE_DIR}" "${WORKSPACE_DIR}"
echo -n "${OPENCLAW_GATEWAY_TOKEN}" > "${STATE_DIR}/.gateway-token"

# Always regenerate config to pick up env var changes
# IMPORTANT: Must always overwrite - never use "if [ ! -f ]" check,
# otherwise stale config on the volume will persist broken settings.
if true; then
  echo "==> Generating config..."
  MODEL="${OPENCLAW_DEFAULT_MODEL:-anthropic/claude-sonnet-4-6}"

  # Build optional sections
  TELEGRAM_BLOCK=""
  if [ -n "${TELEGRAM_BOT_TOKEN:-}" ]; then
    TELEGRAM_BLOCK=$(cat <<TGEOF
  "channels": {
    "telegram": {
      "botToken": "${TELEGRAM_BOT_TOKEN}",
      "dmPolicy": "${TELEGRAM_DM_POLICY:-allowlist}",
      "allowFrom": [${TELEGRAM_ALLOW_FROM:-}]
    }
  },
TGEOF
    )
  fi

  MCP_BLOCK=""
  if [ -n "${AIRTABLE_API_KEY:-}" ]; then
    MCP_BLOCK=$(cat <<MCPEOF
  "mcp": {
    "servers": {
      "airtable": {
        "command": "npx",
        "args": ["-y", "@airtable/mcp-server"],
        "env": {
          "AIRTABLE_API_KEY": "${AIRTABLE_API_KEY}"
        }
      }
    }
  },
MCPEOF
    )
  fi

  cat > "${CONFIG_FILE}" <<EOF
{
  "gateway": {
    "mode": "local", "bind": "lan", "port": 18789,
    "trustedProxies": ["127.0.0.1", "::1"],
    "auth": { "token": "${OPENCLAW_GATEWAY_TOKEN}" },
    "controlUi": { "allowedOrigins": ["*"], "dangerouslyDisableDeviceAuth": true }
  },
  "agents": {
    "defaults": {
      "model": { "primary": "${MODEL}" },
      "workspace": "${WORKSPACE_DIR}",
      "maxConcurrent": 4,
      "subagents": { "maxConcurrent": 8 }
    }
  },
  "tools": {
    "web": { "search": {"enabled":true}, "fetch": {"enabled":true} }
  },
  ${MCP_BLOCK}
  ${TELEGRAM_BLOCK}
  "_generated": true
}
EOF
fi

# Save API keys to persistent .env
touch "${STATE_DIR}/.env"
[ -n "${ANTHROPIC_API_KEY:-}" ] && \
  grep -q "ANTHROPIC_API_KEY" "${STATE_DIR}/.env" 2>/dev/null || \
  echo "ANTHROPIC_API_KEY=${ANTHROPIC_API_KEY:-}" >> "${STATE_DIR}/.env"
[ -n "${OPENAI_API_KEY:-}" ] && \
  grep -q "OPENAI_API_KEY" "${STATE_DIR}/.env" 2>/dev/null || \
  echo "OPENAI_API_KEY=${OPENAI_API_KEY:-}" >> "${STATE_DIR}/.env"

[ -f "${STATE_DIR}/.env" ] && { set -a; source "${STATE_DIR}/.env"; set +a; }

echo ""
echo "==========================================="
echo "  OpenClaw on Railway"
echo "==========================================="
echo "  Model: ${OPENCLAW_DEFAULT_MODEL:-anthropic/claude-sonnet-4-6}"
echo "  GATEWAY TOKEN (copy this!):"
echo "  ${OPENCLAW_GATEWAY_TOKEN}"
echo "==========================================="

# CRITICAL: Export gateway token so embedded agents can connect back
export OPENCLAW_STATE_DIR="${STATE_DIR}"
export OPENCLAW_WORKSPACE_DIR="${WORKSPACE_DIR}"
export OPENCLAW_GATEWAY_TOKEN="${OPENCLAW_GATEWAY_TOKEN}"
export HOME="/root"

# Configure Anthropic setup-token auth if provided
if [ -n "${ANTHROPIC_SETUP_TOKEN:-}" ]; then
  echo "==> Configuring Anthropic setup-token auth..."
  AUTH_DIR="${STATE_DIR}/agents/main/agent"
  mkdir -p "${AUTH_DIR}"
  cat > "${AUTH_DIR}/auth-profiles.json" <<AUTHEOF
{
  "version": 1,
  "profiles": {
    "anthropic:setup-token": {
      "type": "token",
      "provider": "anthropic",
      "token": "${ANTHROPIC_SETUP_TOKEN}"
    }
  },
  "order": {
    "anthropic": ["anthropic:setup-token"]
  }
}
AUTHEOF
  echo "==> Auth profile written to ${AUTH_DIR}/auth-profiles.json"
fi

# Reverse proxy: Railway $PORT -> gateway 18789
# Strips forwarded headers so gateway sees requests as local/trusted
cat > /tmp/proxy.js <<'PROXYEOF'
const http = require("http");
const PORT = process.env.PORT || 8080;
const server = http.createServer((req, res) => {
  if (req.url === "/healthz") {
    res.writeHead(200,{"Content-Type":"application/json"});
    return res.end(JSON.stringify({ok:true}));
  }
  const h = Object.assign({},req.headers);
  delete h["x-forwarded-for"]; delete h["x-forwarded-proto"];
  delete h["x-forwarded-host"]; delete h["x-forwarded-port"];
  const o = {hostname:"127.0.0.1",port:18789,
    path:req.url,method:req.method,headers:h};
  const p = http.request(o, r => {
    res.writeHead(r.statusCode,r.headers);
    r.pipe(res,{end:true});
  });
  p.on("error", () => {
    if(!res.headersSent){res.writeHead(502,{"Content-Type":"text/html"});
    res.end("<h2>Starting up...</h2><p>Refresh in a moment.</p>");}
  });
  req.pipe(p,{end:true});
});
server.on("upgrade",(req,socket)=>{
  const h2=Object.assign({},req.headers);
  delete h2["x-forwarded-for"]; delete h2["x-forwarded-proto"];
  delete h2["x-forwarded-host"]; delete h2["x-forwarded-port"];
  const o={hostname:"127.0.0.1",port:18789,
    path:req.url,method:req.method,headers:h2};
  const p=http.request(o);
  p.on("upgrade",(r,s)=>{
    socket.write("HTTP/1.1 101 Switching Protocols\r\n"
      +Object.entries(r.headers).map(([k,v])=>k+": "+v).join("\r\n")
      +"\r\n\r\n");
    s.pipe(socket);socket.pipe(s);
  });
  p.on("error",()=>socket.end());
  p.end();
});
server.listen(PORT,"0.0.0.0",()=>
  console.log("Proxy: "+PORT+" -> 18789"));
PROXYEOF

node /tmp/proxy.js &
echo "==> Starting OpenClaw gateway..."
exec openclaw gateway \
  --port 18789 --bind lan \
  --token "${OPENCLAW_GATEWAY_TOKEN}" \
  --allow-unconfigured

4 Add Persistent Volume

  1. Click your service → Settings tab (or Volumes section)
  2. Click Add Volume
  3. Mount path: /data
  4. Size: 1 GB is plenty
  5. Click Save
Why this matters: Without a volume, every redeploy wipes the gateway token, auth profiles, and config. The volume persists data at /data/.openclaw across deploys.

5 Deploy & Verify

  1. After adding env vars and volume, Railway will auto-redeploy
  2. Wait for status to show SUCCESS (1-3 minutes)
  3. Railway auto-assigns a domain like your-service-production-xxxx.up.railway.app
  4. Test health: visit https://your-domain.up.railway.app/healthz — should return {"ok":true}
  5. Open web UI: https://your-domain.up.railway.app/
  6. Open chat: https://your-domain.up.railway.app/chat

6 Get Your Gateway Token

You need the gateway token to connect the web UI:

  1. If you set OPENCLAW_GATEWAY_TOKEN as an env var, you already know it
  2. If auto-generated, check the Deploy Logs in Railway — it prints:

Troubleshooting (Known Fixes Already Applied)

This template has all these fixes baked in. Listed here for reference if you customize:

Fix 1: "Invalid input - bind"

Cause: Using "bind": "all" in config

Fix: Valid values are: auto, lan, loopback, custom, tailnet. This template uses lan.

Fix 2: "unknown option --state-dir"

Cause: --state-dir and --workspace-dir are not valid CLI flags

Fix: Use env vars OPENCLAW_STATE_DIR and OPENCLAW_WORKSPACE_DIR instead. Use --allow-unconfigured flag.

Fix 3: "Proxy headers detected from untrusted address"

Cause: Railway's load balancer adds forwarded headers that OpenClaw's gateway doesn't trust

Fix: Two things: (1) Add "trustedProxies": ["127.0.0.1", "::1"] to gateway config. (2) The Node.js reverse proxy strips x-forwarded-* headers before forwarding to the gateway.

Fix 4: "pairing required" for browser connections

Cause: OpenClaw requires device pairing even with valid token for remote browser access

Fix: Add "dangerouslyDisableDeviceAuth": true to gateway.controlUi config.

Fix 5: "No API key found for provider anthropic"

Cause: auth-profiles.json written with wrong format (nested credentials object or missing version)

Fix: Must have "version": 1 at top level. Credentials (type, provider, token) go directly in each profile object, not nested.

Fix 6: "pairing required" for internal agent tools

Cause: Embedded agent connecting back to gateway failed because OPENCLAW_GATEWAY_TOKEN wasn't exported

Fix: export OPENCLAW_GATEWAY_TOKEN before starting the gateway.

Fix 7: Stale config on persistent volume

Cause: First deploy wrote broken config to volume; subsequent deploys read the stale broken config

Fix: Always regenerate config on every boot (the if true; then pattern). Never use if [ ! -f config ]; then.

Fix 8: Telegram "Unrecognized key" errors

Cause: Using wrong field names like "dm", "agent", "allowlist"

Fix: Correct fields are "dmPolicy", "allowFrom". No "agent" or "dm" wrapper object. Schema is strict.

Architecture Summary


  Railway ($PORT)          Node.js Proxy           OpenClaw Gateway
  ┌──────────────┐    ┌───────────────────┐    ┌──────────────────┐
  │ HTTPS in     │───>│ Strip x-forwarded  │───>│ ws://0.0.0.0:    │
  │ (Railway LB) │    │ headers, forward   │    │ 18789            │
  │              │<───│ to localhost:18789 │<───│                  │
  └──────────────┘    └───────────────────┘    └──────────────────┘
       Port 8080         /healthz endpoint       Auth + WebSocket
                         WebSocket upgrade       Telegram bot
                                                 Canvas, Chat UI

The proxy exists because Railway terminates TLS and adds forwarded headers that OpenClaw doesn't trust. The proxy strips those headers and forwards everything (including WebSocket upgrades) to the gateway on localhost.
